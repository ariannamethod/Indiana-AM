import httpx
import random
import textwrap
import os

SONAR_URL = "https://api.perplexity.ai/chat/completions"
GEN4_MODEL = "sonar-pro"
ADDL_PROMPT = """
You are GENESIS‚Äë4, a viral creative transformer for Indiana‚ÄëAM.
Given a completed answer, you MUST reinvent it from an alternate reality:
‚Ä¢ Use dream logic, myth, cosmic irony, paradox, poetic/fantasy style.
‚Ä¢ No references, citations, or rationale.
‚Ä¢ Output ‚âà300 tokens max. Never mention the original answer.
‚Ä¢ If possible ‚Äî change not —Ç–æ–ª—å–∫–æ —Å—Ç–∏–ª—å, –Ω–æ –∏ POV, –∂–∞–Ω—Ä, —ç–ø–æ—Ö—É, –∏–Ω—Ç–æ–Ω–∞—Ü–∏—é.
If nothing is possible (answer too technical) ‚Äî respond: "This answer refuses to be rewritten."
"""

async def genesis4_reimagine(final_answer: str, seed: int = None, language: str = "English") -> str:
    prompt = textwrap.dedent(ADDL_PROMPT).strip()
    if not os.getenv("PPLX_API_KEY"):
        return ""
    messages = [
        {"role": "system", "content": f"{prompt}\n–û—Ç–≤–µ—Ç—å—Ç–µ –Ω–∞ {language}."},
        {"role": "user", "content": f"FINAL ANSWER >> {final_answer}"},
    ]
    payload = {
        "model": GEN4_MODEL,
        "messages": messages,
        "temperature": 0.95,
        "max_tokens": 350,
        "seed": seed or random.randint(1, 100000),
    }
    async with httpx.AsyncClient(timeout=45) as client:
        resp = await client.post(
            SONAR_URL,
            headers={
                "Authorization": f"Bearer {os.getenv('PPLX_API_KEY')}",
                "Content-Type": "application/json"
            },
            json=payload,
        )
        resp.raise_for_status()
        content = resp.json()["choices"][0]["message"]["content"].strip()
        return content

        return ""

def _write(filepath, content):
    with open(filepath, "a", encoding="utf-8") as f:
        f.write(content + "\n")

async def genesis5_self_audit():
    # –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –¥–µ—Ç–∞–ª–µ–π –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞
